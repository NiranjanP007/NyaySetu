CREATE SCHEMA "public";
CREATE TABLE "CASE" (
	"case_no" varchar(15),
	"type" varchar(20) NOT NULL,
	"Category" varchar(30) NOT NULL,
	"Jurisdiction_Type" varchar(30) NOT NULL,
	"Name" varchar(100) NOT NULL,
	"Description" text,
	"curr_status" varchar(20) NOT NULL,
	CONSTRAINT "pk_case" PRIMARY KEY("case_no")
);
CREATE TABLE "CASE_STATUS" (
	"Status_ID" varchar(15),
	"case_no" varchar(15) NOT NULL,
	"status" varchar(20) NOT NULL,
	"curr_stage" varchar(30) NOT NULL,
	"is_dormant" boolean NOT NULL,
	"dormant_reason" text,
	CONSTRAINT "pk_case_status" PRIMARY KEY("Status_ID")
);
CREATE TABLE "COURT" (
	"ID" integer,
	"Location" text NOT NULL,
	"Level" varchar(20) NOT NULL,
	"Pin_Code" integer NOT NULL,
	"Email" varchar(50) NOT NULL CONSTRAINT "uq_court_email" UNIQUE,
	"Phone_No" varchar(15) NOT NULL CONSTRAINT "uq_court_phone" UNIQUE,
	CONSTRAINT "pk_court" PRIMARY KEY("ID")
);
CREATE TABLE "DOCUMENT_REPO" (
	"ID" varchar(20),
	"Name" varchar(50) NOT NULL,
	"Type" varchar(20) NOT NULL,
	"Link" text NOT NULL,
	"BAR_Registration_No" varchar(20),
	"case_no" varchar(15),
	CONSTRAINT "pk_document_repo" PRIMARY KEY("ID")
);
CREATE TABLE "EVIDENCE" (
	"ID" integer,
	"Type" varchar(20) NOT NULL,
	"Description" text,
	"Submission_date" date NOT NULL,
	"Verified_Status" boolean NOT NULL,
	"case_no" varchar(15) NOT NULL,
	CONSTRAINT "pk_evidence" PRIMARY KEY("ID")
);
CREATE TABLE "FILING_DETAILS" (
	"ID" varchar(15),
	"Registration_date" date NOT NULL,
	"filing_mode" varchar(50) NOT NULL,
	"case_no" varchar(15) NOT NULL,
	CONSTRAINT "pk_filing_details" PRIMARY KEY("ID")
);
CREATE TABLE "FORMS" (
	"judge_id" integer,
	"panel_id" integer,
	CONSTRAINT "pk_forms" PRIMARY KEY("judge_id","panel_id")
);
CREATE TABLE "HANDLED_BY" (
	"BAR_Registration_No" varchar(20),
	"case_no" varchar(15),
	CONSTRAINT "pk_handled_by" PRIMARY KEY("BAR_Registration_No","case_no")
);
CREATE TABLE "HANDLES" (
	"court_id" integer,
	"case_no" varchar(15),
	CONSTRAINT "pk_handles" PRIMARY KEY("court_id","case_no")
);
CREATE TABLE "HEARD_BY" (
	"hearing_no" integer,
	"panel_id" integer,
	CONSTRAINT "pk_heard_by" PRIMARY KEY("hearing_no","panel_id")
);
CREATE TABLE "HEARING" (
	"Hearing_No" integer,
	"case_no" varchar(15) NOT NULL,
	"court_id" integer NOT NULL,
	"Type" varchar(20) NOT NULL,
	"Status" varchar(20) NOT NULL,
	CONSTRAINT "pk_hearing" PRIMARY KEY("Hearing_No")
);
CREATE TABLE "HEARING_TRANSCRIPT" (
	"ID" integer,
	"Hearing_no" integer,
	"Text" text NOT NULL,
	"recorded_by" integer NOT NULL,
	"court_id" integer NOT NULL,
	"Time_stamp" time NOT NULL,
	CONSTRAINT "pk_hearing_transcript" PRIMARY KEY("Hearing_no","ID")
);
CREATE TABLE "IS_PRECEDENT" (
	"case_no" varchar(15),
	"precedent_case_no" varchar(15),
	CONSTRAINT "pk_is_precedent" PRIMARY KEY("case_no","precedent_case_no")
);
CREATE TABLE "JUDGE" (
	"ID" integer,
	"Name" varchar(100) NOT NULL,
	"Gender" varchar(15) NOT NULL,
	"DOB" date NOT NULL,
	"Email" varchar(50) NOT NULL CONSTRAINT "uq_judge_email" UNIQUE,
	"Status" varchar(20) NOT NULL,
	"Phone_No" varchar(15) NOT NULL CONSTRAINT "uq_judge_phone" UNIQUE,
	"Address" text,
	"Position" varchar(30) NOT NULL,
	"Qualification" varchar(50) NOT NULL,
	"Experience" varchar(50),
	CONSTRAINT "pk_judge" PRIMARY KEY("ID")
);
CREATE TABLE "JUDGEMENT" (
	"ID" varchar(25),
	"Decision_date" date NOT NULL,
	"verdict" text NOT NULL,
	"compensations" text,
	"sentence_length_punishment" text,
	"remark" text,
	"case_no" varchar(15) NOT NULL,
	"panel_id" integer,
	CONSTRAINT "pk_judgement" PRIMARY KEY("ID")
);
CREATE TABLE "LAWYER" (
	"BAR_Registration_No" varchar(20),
	"Qualification" varchar(100) NOT NULL,
	"Specialization" varchar(100) NOT NULL,
	"Phone_No" varchar(15) NOT NULL CONSTRAINT "uq_lawyer_phone" UNIQUE,
	"Email" varchar(50) NOT NULL CONSTRAINT "uq_lawyer_email" UNIQUE,
	"Office_Address" text,
	"Experience" text,
	"Status" varchar(20) NOT NULL,
	CONSTRAINT "pk_lawyer" PRIMARY KEY("BAR_Registration_No")
);
CREATE TABLE "LITIGANTS" (
	"ID" varchar(20),
	"Name" varchar(100) NOT NULL,
	"Type" varchar(30) NOT NULL,
	"Address" text,
	"Phone" varchar(15) NOT NULL CONSTRAINT "uq_litigants_phone" UNIQUE,
	"Email" varchar(50) NOT NULL CONSTRAINT "uq_litigants_email" UNIQUE,
	"Gender" varchar(10),
	CONSTRAINT "pk_litigants" PRIMARY KEY("ID")
);
CREATE TABLE "PANEL" (
	"ID" integer,
	"type" varchar(30) NOT NULL,
	"form_date" date NOT NULL,
	"dissolve_date" date,
	CONSTRAINT "pk_panel" PRIMARY KEY("ID")
);
CREATE TABLE "PARTICIPATES_IN" (
	"litigant_id" varchar(20),
	"case_no" varchar(15),
	CONSTRAINT "pk_participates_in" PRIMARY KEY("litigant_id","case_no")
);
CREATE TABLE "query_logs" (
	"id" serial PRIMARY KEY,
	"query" text NOT NULL,
	"status" varchar(255) NOT NULL,
	"execution_time_ms" integer NOT NULL,
	"executed_at" timestamp DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE "saved_queries" (
	"id" serial PRIMARY KEY,
	"name" varchar(255) NOT NULL,
	"query" text NOT NULL,
	"created_at" timestamp DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE "WARRANT" (
	"ID" varchar(15),
	"Type" varchar(20) NOT NULL,
	"Issue_Date" timestamp NOT NULL,
	"Exp_date" timestamp,
	"Status" varchar(20) NOT NULL,
	"case_no" varchar(15) NOT NULL,
	CONSTRAINT "pk_warrant" PRIMARY KEY("ID")
);
CREATE TABLE "WORKS_FOR" (
	"judge_id" integer,
	"court_id" integer,
	CONSTRAINT "pk_works_for" PRIMARY KEY("judge_id","court_id")
);
